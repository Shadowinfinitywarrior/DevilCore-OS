1.2 Design Principles

text

1. MINIMAL FOOTPRINT    — Target < 500MB installed, < 128MB RAM idle
2. PRIVACY BY DEFAULT   — All traffic anonymized, no telemetry, ever
3. SECURITY FIRST       — Mandatory access control, encrypted everything
4. LINUX COMPATIBLE     — Read ext2/3/4, run ELF binaries (compatibility layer)
5. HACKER FRIENDLY      — Built-in tools for ethical hacking / pentesting
6. FROM SCRATCH         — Every component written fresh, no borrowed kernels
7. MODULAR KERNEL       — Hybrid kernel (monolithic core + loadable modules)

1.3 Target Specifications

text

Architecture:       x86_64 (primary), ARM64 (future)
Boot:               UEFI (primary), Legacy BIOS (fallback)
Min RAM:            256 MB
Recommended RAM:    2 GB+
Min Storage:        2 GB
Kernel Type:        Hybrid (monolithic core + loadable modules)
GUI:                Custom compositor (Wayland-inspired protocol)
File Systems:       DevilFS (native), ext2/3/4 (read/write), FAT32
Networking:         Full TCP/IP stack with built-in Tor/VPN

PART 3: BOOTLOADER (DevilBoot)
3.1 Overview

DevilBoot is a two-stage bootloader:

    Stage 1: Fits in MBR (512 bytes) or UEFI entry. Sets up basics and loads Stage 2.
    Stage 2: Enters protected mode → long mode (64-bit), loads kernel ELF into memory.

3.2 Stage 1 — MBR Boot Sector (BIOS)

nasm

; bootloader/stage1/mbr.asm
; Stage 1 MBR Bootloader for DevilCore
; Loaded at 0x7C00 by BIOS

[BITS 16]
[ORG 0x7C00]

; ============================================
; Constants
; ============================================
STAGE2_SEGMENT  equ 0x1000
STAGE2_OFFSET   equ 0x0000
STAGE2_SECTORS  equ 32          ; 16KB for stage 2
STAGE2_START    equ 1           ; LBA sector 1 (after MBR)

; ============================================
; Entry Point
; ============================================
start:
    ; Set up segments
    cli
    xor     ax, ax
    mov     ds, ax
    mov     es, ax
    mov     ss, ax
    mov     sp, 0x7C00          ; Stack below bootloader
    sti

    ; Save boot drive number
    mov     [boot_drive], dl

    ; Print boot message
    mov     si, msg_boot
    call    print_string

    ; ============================================
    ; Load Stage 2 from disk
    ; ============================================
    mov     si, msg_loading
    call    print_string

    ; Reset disk system
    xor     ax, ax
    mov     dl, [boot_drive]
    int     0x13
    jc      disk_error

    ; Load Stage 2 using LBA (INT 13h extensions)
    mov     si, dap
    mov     ah, 0x42
    mov     dl, [boot_drive]
    int     0x13
    jc      disk_error

    ; Print success
    mov     si, msg_loaded
    call    print_string

    ; ============================================
    ; Enable A20 Line
    ; ============================================
    call    enable_a20

    ; ============================================
    ; Jump to Stage 2
    ; ============================================
    mov     dl, [boot_drive]    ; Pass drive number
    jmp     STAGE2_SEGMENT:STAGE2_OFFSET

; ============================================
; Enable A20 Line (Fast A20)
; ============================================
enable_a20:
    in      al, 0x92
    test    al, 2
    jnz     .done
    or      al, 2
    and     al, 0xFE
    out     0x92, al
.done:
    ret

; ============================================
; Print String (SI = pointer)
; ============================================
print_string:
    pusha
.loop:
    lodsb
    test    al, al
    jz      .done
    mov     ah, 0x0E
    mov     bh, 0
    mov     bl, 0x07
    int     0x10
    jmp     .loop
.done:
    popa
    ret

; ============================================
; Disk Error Handler
; ============================================
disk_error:
    mov     si, msg_disk_err
    call    print_string
    cli
    hlt

; ============================================
; Data
; ============================================
boot_drive:     db 0

; Disk Address Packet for LBA read
dap:
    db  0x10                    ; Size of DAP
    db  0                       ; Reserved
    dw  STAGE2_SECTORS          ; Number of sectors
    dw  STAGE2_OFFSET           ; Offset
    dw  STAGE2_SEGMENT          ; Segment
    dq  STAGE2_START            ; LBA start

; Messages
msg_boot:       db "[DevilBoot] DevilCore OS Bootloader v1.0", 13, 10, 0
msg_loading:    db "[DevilBoot] Loading Stage 2...", 13, 10, 0
msg_loaded:     db "[DevilBoot] Stage 2 loaded. Jumping...", 13, 10, 0
msg_disk_err:   db "[DevilBoot] DISK ERROR!", 13, 10, 0

; ============================================
; Boot Signature
; ============================================
times 510 - ($ - $$) db 0
dw 0xAA55

3.3 Stage 2 — Protected Mode & Long Mode Setup

nasm

; bootloader/stage2/loader.asm
; Stage 2: Set up protected mode, long mode, load kernel

[BITS 16]
[ORG 0x0000]                    ; Loaded at STAGE2_SEGMENT:0x0000

section .text

stage2_start:
    ; Set up segments for this location
    mov     ax, cs
    mov     ds, ax
    mov     es, ax

    ; Save boot drive
    mov     [boot_drive], dl

    ; Print stage 2 message
    mov     si, msg_stage2
    call    print_string_16

    ; ============================================
    ; Get Memory Map (E820)
    ; ============================================
    call    get_memory_map

    ; ============================================
    ; Load Kernel from disk
    ; ============================================
    call    load_kernel

    ; ============================================
    ; Set Video Mode (VESA)
    ; ============================================
    call    set_vesa_mode

    ; ============================================
    ; Enter Protected Mode
    ; ============================================
    cli
    lgdt    [gdt_descriptor]

    mov     eax, cr0
    or      eax, 1              ; Set PE bit
    mov     cr0, eax

    jmp     CODE_SEG:protected_mode_entry

; ============================================
; Get Memory Map using INT 15h, E820
; ============================================
get_memory_map:
    mov     si, msg_memmap
    call    print_string_16

    mov     di, 0x8000          ; Store memory map at 0x8000
    xor     ebx, ebx
    xor     bp, bp              ; Entry count
    mov     edx, 0x534D4150     ; 'SMAP'

.loop:
    mov     eax, 0xE820
    mov     ecx, 24
    int     0x15
    jc      .done

    add     di, 24
    inc     bp
    test    ebx, ebx
    jnz     .loop

.done:
    mov     [0x7FF0], bp        ; Store entry count
    ret

; ============================================
; Load Kernel ELF from disk
; Kernel starts at LBA 33 (after bootloader)
; Load to 0x100000 (1MB) temporarily
; ============================================
load_kernel:
    mov     si, msg_kernel
    call    print_string_16

    ; We'll load kernel in chunks (64KB at a time)
    ; to a temporary buffer then copy to high memory
    ; For simplicity, load to 0x20000-0x90000 first

    mov     word [dap_sectors], 128     ; 64KB
    mov     word [dap_offset], 0x0000
    mov     word [dap_segment], 0x2000
    mov     dword [dap_lba_low], 33     ; Kernel starts at LBA 33
    mov     dword [dap_lba_high], 0

    ; Load 512KB of kernel (8 chunks of 64KB)
    mov     cx, 8
    mov     bx, 0x2000          ; Starting segment

.load_loop:
    push    cx

    mov     si, dap_packet
    mov     ah, 0x42
    mov     dl, [boot_drive]
    int     0x13
    jc      .load_error

    ; Advance segment by 0x1000 (64KB)
    add     word [dap_segment], 0x1000
    ; Advance LBA by 128 sectors
    add     dword [dap_lba_low], 128

    pop     cx
    loop    .load_loop

    mov     si, msg_kernel_ok
    call    print_string_16
    ret

.load_error:
    mov     si, msg_kernel_err
    call    print_string_16
    cli
    hlt

; DAP for kernel loading
dap_packet:
    db      0x10                ; Size
    db      0                   ; Reserved
dap_sectors:
    dw      128                 ; Sectors to read
dap_offset:
    dw      0x0000              ; Offset
dap_segment:
    dw      0x2000              ; Segment
dap_lba_low:
    dd      33                  ; LBA low 32 bits
dap_lba_high:
    dd      0                   ; LBA high 32 bits

; ============================================
; Set VESA Mode (1024x768x32)
; ============================================
set_vesa_mode:
    mov     si, msg_vesa
    call    print_string_16

    ; Get VESA info for mode 0x118 (1024x768x32bpp)
    mov     ax, 0x4F01
    mov     cx, 0x118
    mov     di, 0x7E00          ; Mode info buffer
    int     0x10
    cmp     ax, 0x004F
    jne     .vesa_fail

    ; Set the mode
    mov     ax, 0x4F02
    mov     bx, 0x4118          ; Mode 0x118 + linear framebuffer bit
    int     0x10
    cmp     ax, 0x004F
    jne     .vesa_fail

    ret

.vesa_fail:
    ; Fall back to VGA 320x200
    mov     ax, 0x0013
    int     0x10
    ret

; ============================================
; Print String 16-bit
; ============================================
print_string_16:
    pusha
.loop:
    lodsb
    test    al, al
    jz      .done
    mov     ah, 0x0E
    int     0x10
    jmp     .loop
.done:
    popa
    ret

; ============================================
; Messages
; ============================================
msg_stage2:     db "[DevilBoot] Stage 2 loaded.", 13, 10, 0
msg_memmap:     db "[DevilBoot] Getting memory map...", 13, 10, 0
msg_kernel:     db "[DevilBoot] Loading kernel...", 13, 10, 0
msg_kernel_ok:  db "[DevilBoot] Kernel loaded OK.", 13, 10, 0
msg_kernel_err: db "[DevilBoot] Kernel load FAILED!", 13, 10, 0
msg_vesa:       db "[DevilBoot] Setting video mode...", 13, 10, 0

boot_drive:     db 0

; ============================================
; GDT for Protected Mode
; ============================================
gdt_start:
    ; Null descriptor
    dq 0

gdt_code:
    ; Code segment: base=0, limit=4GB, 32-bit, executable
    dw 0xFFFF       ; Limit low
    dw 0x0000       ; Base low
    db 0x00         ; Base middle
    db 10011010b    ; Access: present, ring 0, code, readable
    db 11001111b    ; Flags: 4KB granularity, 32-bit + limit high
    db 0x00         ; Base high

gdt_data:
    ; Data segment: base=0, limit=4GB, 32-bit, writable
    dw 0xFFFF
    dw 0x0000
    db 0x00
    db 10010010b    ; Access: present, ring 0, data, writable
    db 11001111b
    db 0x00

gdt_end:

gdt_descriptor:
    dw gdt_end - gdt_start - 1  ; Size
    dd gdt_start                 ; Address (will need fixup)

CODE_SEG equ gdt_code - gdt_start
DATA_SEG equ gdt_data - gdt_start

; ============================================
; Protected Mode Entry
; ============================================
[BITS 32]
protected_mode_entry:
    ; Set up data segments
    mov     ax, DATA_SEG
    mov     ds, ax
    mov     es, ax
    mov     fs, ax
    mov     gs, ax
    mov     ss, ax
    mov     esp, 0x90000        ; Stack at 576KB

    ; ============================================
    ; Copy kernel from 0x20000 to 0x100000 (1MB)
    ; ============================================
    mov     esi, 0x20000        ; Source
    mov     edi, 0x100000       ; Destination (1MB)
    mov     ecx, 131072         ; 512KB / 4 = 131072 dwords
    rep     movsd

    ; ============================================
    ; Set up page tables for long mode
    ; ============================================
    call    setup_page_tables

    ; ============================================
    ; Enable Long Mode
    ; ============================================

    ; Enable PAE
    mov     eax, cr4
    or      eax, (1 << 5)       ; PAE bit
    mov     cr4, eax

    ; Set PML4 address
    mov     eax, 0x70000        ; PML4 table at 0x70000
    mov     cr3, eax

    ; Enable long mode in EFER MSR
    mov     ecx, 0xC0000080     ; EFER MSR
    rdmsr
    or      eax, (1 << 8)       ; LME bit
    wrmsr

    ; Enable paging
    mov     eax, cr0
    or      eax, (1 << 31)      ; PG bit
    mov     cr0, eax

    ; Load 64-bit GDT
    lgdt    [gdt64_descriptor]

    ; Far jump to 64-bit code
    jmp     0x08:long_mode_entry

; ============================================
; Set up Identity-Mapped Page Tables
; Map first 4GB of memory
; ============================================
setup_page_tables:
    ; Clear page table area (0x70000 - 0x76000)
    mov     edi, 0x70000
    xor     eax, eax
    mov     ecx, 0x6000 / 4
    rep     stosd

    ; PML4[0] -> PDPT at 0x71000
    mov     dword [0x70000], 0x71003    ; Present + Writable

    ; PDPT[0] -> PD at 0x72000
    ; PDPT[1] -> PD at 0x73000
    ; PDPT[2] -> PD at 0x74000
    ; PDPT[3] -> PD at 0x75000
    mov     dword [0x71000], 0x72003
    mov     dword [0x71008], 0x73003
    mov     dword [0x71010], 0x74003
    mov     dword [0x71018], 0x75003

    ; Map 4GB using 2MB pages
    mov     edi, 0x72000        ; First PD
    mov     eax, 0x00000083     ; Present + Writable + 2MB page
    mov     ecx, 2048           ; 4 PDs * 512 entries = 2048

.map_loop:
    mov     [edi], eax
    add     eax, 0x200000       ; Next 2MB
    add     edi, 8
    loop    .map_loop

    ret

; ============================================
; 64-bit GDT
; ============================================
gdt64_start:
    dq 0                        ; Null descriptor

gdt64_code:
    ; 64-bit code segment
    dw 0xFFFF                   ; Limit
    dw 0                        ; Base low
    db 0                        ; Base middle
    db 10011010b                ; Access
    db 10101111b                ; Flags (long mode) + limit
    db 0                        ; Base high

gdt64_data:
    ; 64-bit data segment
    dw 0xFFFF
    dw 0
    db 0
    db 10010010b
    db 10101111b
    db 0

gdt64_end:

gdt64_descriptor:
    dw gdt64_end - gdt64_start - 1
    dd gdt64_start

; ============================================
; Long Mode (64-bit) Entry
; ============================================
[BITS 64]
long_mode_entry:
    ; Set up 64-bit data segments
    mov     ax, 0x10            ; Data segment selector
    mov     ds, ax
    mov     es, ax
    mov     fs, ax
    mov     gs, ax
    mov     ss, ax

    ; Set up stack
    mov     rsp, 0x200000       ; 2MB stack

    ; ============================================
    ; Parse kernel ELF and jump to entry point
    ; ============================================
    ; Kernel is loaded at 0x100000
    ; Parse ELF header to find entry point

    mov     rdi, 0x100000       ; Kernel location
    ; Check ELF magic
    cmp     dword [rdi], 0x464C457F  ; \x7FELF
    jne     .not_elf

    ; Get entry point from ELF header (offset 0x18 in 64-bit ELF)
    mov     rax, [rdi + 0x18]   ; e_entry

    ; Get program header offset
    mov     rbx, [rdi + 0x20]   ; e_phoff
    movzx   rcx, word [rdi + 0x38]  ; e_phnum
    movzx   rdx, word [rdi + 0x36]  ; e_phentsize

    ; Load ELF segments
    lea     rsi, [rdi + rbx]    ; Program header table

.load_segment:
    cmp     dword [rsi], 1      ; PT_LOAD?
    jne     .next_segment

    ; Load this segment
    push    rcx
    push    rsi
    push    rax

    mov     rcx, [rsi + 0x20]   ; p_filesz (file size)
    mov     rdi_dest, [rsi + 0x10]  ; p_vaddr (virtual address)
    mov     rsi_src, 0x100000
    add     rsi_src, [rsi + 0x08]   ; p_offset

    ; Copy segment
    push    rdi
    mov     rdi, [rsi + 0x10]   ; p_vaddr
    mov     rsi, rsi_src
    rep     movsb
    pop     rdi

    pop     rax
    pop     rsi
    pop     rcx

.next_segment:
    add     rsi, rdx            ; Next program header
    loop    .load_segment

    ; ============================================
    ; Prepare boot info structure for kernel
    ; ============================================
    ; RDI = pointer to boot_info structure
    mov     rdi, 0x500          ; Boot info at 0x500

    ; Fill boot_info
    mov     qword [rdi + 0],  0x100000   ; kernel_physical
    mov     qword [rdi + 8],  0x20000    ; memory_map_addr
    mov     qword [rdi + 16], 0x7E00     ; vesa_info_addr
    ; Memory map count stored at 0x7FF0
    movzx   rax, word [0x7FF0]
    mov     qword [rdi + 24], rax        ; memory_map_entries

    ; Get framebuffer info from VESA mode info
    mov     eax, [0x7E00 + 40]  ; Framebuffer physical address
    mov     qword [rdi + 32], rax
    movzx   eax, word [0x7E00 + 18]  ; X resolution
    mov     qword [rdi + 40], rax
    movzx   eax, word [0x7E00 + 20]  ; Y resolution
    mov     qword [rdi + 48], rax
    movzx   eax, byte [0x7E00 + 25]  ; BPP
    mov     qword [rdi + 56], rax

    ; ============================================
    ; Jump to kernel entry point!
    ; ============================================
    mov     rdi, 0x500          ; Boot info pointer in RDI
    mov     rax, [0x100000 + 0x18]  ; ELF entry point
    call    rax

    ; Should never reach here
    cli
    hlt

.not_elf:
    ; Error: not a valid ELF
    cli
    hlt

; Pad to fill sectors
times 16384 - ($ - $$) db 0

3.4 UEFI Boot Entry (Alternative)

C

// bootloader/stage1/uefi_entry.c
// UEFI bootloader entry point for DevilCore

#include <efi.h>
#include <efilib.h>

typedef struct {
    uint64_t framebuffer_addr;
    uint64_t framebuffer_size;
    uint32_t width;
    uint32_t height;
    uint32_t pixels_per_scanline;
    uint32_t bpp;
    EFI_MEMORY_DESCRIPTOR *memory_map;
    uint64_t memory_map_size;
    uint64_t descriptor_size;
    uint64_t descriptor_version;
} BootInfo;

EFI_STATUS EFIAPI efi_main(EFI_HANDLE ImageHandle, EFI_SYSTEM_TABLE *SystemTable) {
    EFI_STATUS status;
    BootInfo *boot_info;

    // Initialize UEFI library
    InitializeLib(ImageHandle, SystemTable);
    Print(L"[DevilBoot] UEFI Bootloader v1.0\r\n");

    // ============================================
    // 1. Set up Graphics Output Protocol (GOP)
    // ============================================
    EFI_GRAPHICS_OUTPUT_PROTOCOL *gop;
    EFI_GUID gop_guid = EFI_GRAPHICS_OUTPUT_PROTOCOL_GUID;

    status = uefi_call_wrapper(BS->LocateProtocol, 3,
        &gop_guid, NULL, (void**)&gop);
    if (EFI_ERROR(status)) {
        Print(L"[DevilBoot] Failed to locate GOP!\r\n");
        return status;
    }

    // Set preferred mode (1024x768 or best available)
    UINTN desired_width = 1920, desired_height = 1080;
    UINTN mode_num = 0;
    BOOLEAN found = FALSE;

    for (UINTN i = 0; i < gop->Mode->MaxMode; i++) {
        EFI_GRAPHICS_OUTPUT_MODE_INFORMATION *info;
        UINTN info_size;
        status = uefi_call_wrapper(gop->QueryMode, 4,
            gop, i, &info_size, &info);
        if (!EFI_ERROR(status)) {
            if (info->HorizontalResolution == desired_width &&
                info->VerticalResolution == desired_height) {
                mode_num = i;
                found = TRUE;
                break;
            }
        }
    }

    if (found) {
        uefi_call_wrapper(gop->SetMode, 2, gop, mode_num);
    }

    Print(L"[DevilBoot] Framebuffer at: 0x%lx\r\n",
        gop->Mode->FrameBufferBase);
    Print(L"[DevilBoot] Resolution: %dx%d\r\n",
        gop->Mode->Info->HorizontalResolution,
        gop->Mode->Info->VerticalResolution);

    // ============================================
    // 2. Load Kernel ELF from disk
    // ============================================
    EFI_FILE_PROTOCOL *root_dir, *kernel_file;
    EFI_LOADED_IMAGE_PROTOCOL *loaded_image;
    EFI_SIMPLE_FILE_SYSTEM_PROTOCOL *fs;

    // Get loaded image protocol
    EFI_GUID lip_guid = EFI_LOADED_IMAGE_PROTOCOL_GUID;
    status = uefi_call_wrapper(BS->HandleProtocol, 3,
        ImageHandle, &lip_guid, (void**)&loaded_image);

    // Get filesystem protocol
    EFI_GUID fs_guid = EFI_SIMPLE_FILE_SYSTEM_PROTOCOL_GUID;
    status = uefi_call_wrapper(BS->HandleProtocol, 3,
        loaded_image->DeviceHandle, &fs_guid, (void**)&fs);

    // Open root directory
    status = uefi_call_wrapper(fs->OpenVolume, 2, fs, &root_dir);

    // Open kernel file
    status = uefi_call_wrapper(root_dir->Open, 5, root_dir,
        &kernel_file, L"\\EFI\\DEVILCORE\\KERNEL.ELF",
        EFI_FILE_MODE_READ, 0);
    if (EFI_ERROR(status)) {
        Print(L"[DevilBoot] Failed to open kernel!\r\n");
        return status;
    }

    // Get file size
    EFI_FILE_INFO *file_info;
    UINTN info_size = sizeof(EFI_FILE_INFO) + 200;
    EFI_GUID fi_guid = EFI_FILE_INFO_ID;
    status = uefi_call_wrapper(BS->AllocatePool, 3,
        EfiLoaderData, info_size, (void**)&file_info);
    status = uefi_call_wrapper(kernel_file->GetInfo, 4,
        kernel_file, &fi_guid, &info_size, file_info);

    UINTN kernel_size = file_info->FileSize;
    Print(L"[DevilBoot] Kernel size: %lu bytes\r\n", kernel_size);

    // Allocate memory for kernel
    void *kernel_buffer;
    status = uefi_call_wrapper(BS->AllocatePool, 3,
        EfiLoaderData, kernel_size, &kernel_buffer);

    // Read kernel
    status = uefi_call_wrapper(kernel_file->Read, 3,
        kernel_file, &kernel_size, kernel_buffer);

    Print(L"[DevilBoot] Kernel loaded at: 0x%lx\r\n", kernel_buffer);

    // ============================================
    // 3. Parse ELF and load segments
    // ============================================
    // (ELF parsing code — load PT_LOAD segments to their p_vaddr)
    typedef struct {
        uint8_t  magic[4];
        uint8_t  class_;
        uint8_t  data;
        uint8_t  version;
        uint8_t  osabi;
        uint64_t padding;
        uint16_t type;
        uint16_t machine;
        uint32_t version2;
        uint64_t entry;
        uint64_t phoff;
        uint64_t shoff;
        uint32_t flags;
        uint16_t ehsize;
        uint16_t phentsize;
        uint16_t phnum;
        uint16_t shentsize;
        uint16_t shnum;
        uint16_t shstrndx;
    } __attribute__((packed)) Elf64_Ehdr;

    typedef struct {
        uint32_t type;
        uint32_t flags;
        uint64_t offset;
        uint64_t vaddr;
        uint64_t paddr;
        uint64_t filesz;
        uint64_t memsz;
        uint64_t align;
    } __attribute__((packed)) Elf64_Phdr;

    Elf64_Ehdr *ehdr = (Elf64_Ehdr*)kernel_buffer;

    // Verify ELF magic
    if (ehdr->magic[0] != 0x7F || ehdr->magic[1] != 'E' ||
        ehdr->magic[2] != 'L'  || ehdr->magic[3] != 'F') {
        Print(L"[DevilBoot] Invalid kernel ELF!\r\n");
        return EFI_INVALID_PARAMETER;
    }

    uint64_t entry_point = ehdr->entry;
    Print(L"[DevilBoot] Kernel entry: 0x%lx\r\n", entry_point);

    // Load segments
    Elf64_Phdr *phdr = (Elf64_Phdr*)((uint8_t*)kernel_buffer + ehdr->phoff);
    for (uint16_t i = 0; i < ehdr->phnum; i++) {
        if (phdr[i].type == 1) { // PT_LOAD
            // Allocate pages at virtual address
            UINTN pages = (phdr[i].memsz + 4095) / 4096;
            EFI_PHYSICAL_ADDRESS paddr = phdr[i].vaddr;
            status = uefi_call_wrapper(BS->AllocatePages, 4,
                AllocateAddress, EfiLoaderData, pages, &paddr);

            // Copy segment data
            uint8_t *src = (uint8_t*)kernel_buffer + phdr[i].offset;
            uint8_t *dst = (uint8_t*)phdr[i].vaddr;
            for (UINTN j = 0; j < phdr[i].filesz; j++)
                dst[j] = src[j];

            // Zero BSS
            for (UINTN j = phdr[i].filesz; j < phdr[i].memsz; j++)
                dst[j] = 0;
        }
    }

    // ============================================
    // 4. Get Memory Map & Exit Boot Services
    // ============================================
    UINTN mmap_size = 0, map_key, desc_size;
    UINT32 desc_version;
    EFI_MEMORY_DESCRIPTOR *mmap = NULL;

    // Get memory map size
    status = uefi_call_wrapper(BS->GetMemoryMap, 5,
        &mmap_size, mmap, &map_key, &desc_size, &desc_version);

    mmap_size += 2 * desc_size; // Extra space
    status = uefi_call_wrapper(BS->AllocatePool, 3,
        EfiLoaderData, mmap_size, (void**)&mmap);

    status = uefi_call_wrapper(BS->GetMemoryMap, 5,
        &mmap_size, mmap, &map_key, &desc_size, &desc_version);

    // ============================================
    // 5. Prepare Boot Info
    // ============================================
    status = uefi_call_wrapper(BS->AllocatePool, 3,
        EfiLoaderData, sizeof(BootInfo), (void**)&boot_info);

    boot_info->framebuffer_addr = gop->Mode->FrameBufferBase;
    boot_info->framebuffer_size = gop->Mode->FrameBufferSize;
    boot_info->width = gop->Mode->Info->HorizontalResolution;
    boot_info->height = gop->Mode->Info->VerticalResolution;
    boot_info->pixels_per_scanline = gop->Mode->Info->PixelsPerScanLine;
    boot_info->bpp = 32;
    boot_info->memory_map = mmap;
    boot_info->memory_map_size = mmap_size;
    boot_info->descriptor_size = desc_size;
    boot_info->descriptor_version = desc_version;

    // ============================================
    // 6. Exit Boot Services
    // ============================================
    status = uefi_call_wrapper(BS->ExitBootServices, 2,
        ImageHandle, map_key);
    if (EFI_ERROR(status)) {
        // Retry with fresh map key
        mmap_size = 0;
        uefi_call_wrapper(BS->GetMemoryMap, 5,
            &mmap_size, NULL, &map_key, &desc_size, &desc_version);
        mmap_size += 2 * desc_size;
        uefi_call_wrapper(BS->AllocatePool, 3,
            EfiLoaderData, mmap_size, (void**)&mmap);
        uefi_call_wrapper(BS->GetMemoryMap, 5,
            &mmap_size, mmap, &map_key, &desc_size, &desc_version);
        uefi_call_wrapper(BS->ExitBootServices, 2,
            ImageHandle, map_key);
    }

    // ============================================
    // 7. Jump to Kernel!
    // ============================================
    typedef void (*KernelEntry)(BootInfo*);
    KernelEntry kernel_main = (KernelEntry)entry_point;
    kernel_main(boot_info);

    // Should never reach here
    while(1) __asm__ volatile("hlt");
    return EFI_SUCCESS;
}

section .multiboot
align 8
mb_header_start:
    dd  0xE85250D6              ; Magic
    dd  0                       ; Architecture (i386/x86_64)
    dd  mb_header_end - mb_header_start  ; Header length
    dd  -(0xE85250D6 + 0 + (mb_header_end - mb_header_start))  ; Checksum
    ; End tag
    dw  0                       ; Type
    dw  0                       ; Flags
    dd  8                       ; Size
mb_header_end:

section .bss
align 16
stack_bottom:
    resb 65536                  ; 64KB kernel stack
stack_top:

section .text
_start:
    ; RDI already contains boot_info pointer from bootloader
    mov     rsp, stack_top      ; Set up kernel stack

    ; Clear direction flag
    cld

    ; Zero out BSS section
    extern __bss_start
    extern __bss_end
    mov     rdi, __bss_start
    mov     rcx, __bss_end
    sub     rcx, rdi
    shr     rcx, 3              ; Divide by 8 (qwords)
    xor     rax, rax
    rep     stosq

    ; Restore RDI (boot_info pointer)
    ; (It was saved or we reload it)

    ; Call kernel_main(boot_info)
    ; RDI still has boot_info from bootloader
    call    kernel_main

    ; Kernel returned — halt
    cli
.hang:
    hlt
    jmp     .hang

4.4 GDT (Global Descriptor Table)

C

// kernel/arch/x86_64/gdt.c

#include <devil/types.h>
#include "gdt.h"

// GDT entry structure
typedef struct {
    uint16_t limit_low;
    uint16_t base_low;
    uint8_t  base_middle;
    uint8_t  access;
    uint8_t  granularity;
    uint8_t  base_high;
} PACKED GDTEntry;

// 64-bit TSS descriptor (16 bytes)
typedef struct {
    uint16_t limit_low;
    uint16_t base_low;
    uint8_t  base_middle;
    uint8_t  access;
    uint8_t  granularity;
    uint8_t  base_high;
    uint32_t base_upper;
    uint32_t reserved;
} PACKED TSSDescriptor;

// GDT pointer
typedef struct {
    uint16_t limit;
    uint64_t base;
} PACKED GDTPointer;

// TSS structure
typedef struct {
    uint32_t reserved0;
    uint64_t rsp0;          // Kernel stack for ring 0
    uint64_t rsp1;
    uint64_t rsp2;
    uint64_t reserved1;
    uint64_t ist1;          // Interrupt Stack Table
    uint64_t ist2;
    uint64_t ist3;
    uint64_t ist4;
    uint64_t ist5;
    uint64_t ist6;
    uint64_t ist7;
    uint64_t reserved2;
    uint16_t reserved3;
    uint16_t iopb_offset;
} PACKED TSS;

// GDT entries:
// 0: Null
// 1: Kernel Code (0x08)
// 2: Kernel Data (0x10)
// 3: User Code   (0x18) -- with DPL=3
// 4: User Data   (0x20) -- with DPL=3
// 5-6: TSS       (0x28) -- 16 bytes

#define GDT_ENTRIES 7
static GDTEntry gdt[GDT_ENTRIES] ALIGNED(16);
static GDTPointer gdt_ptr;
static TSS tss ALIGNED(16);

static void gdt_set_entry(int index, uint32_t base, uint32_t limit,
                          uint8_t access, uint8_t gran) {
    gdt[index].base_low    = base & 0xFFFF;
    gdt[index].base_middle = (base >> 16) & 0xFF;
    gdt[index].base_high   = (base >> 24) & 0xFF;
    gdt[index].limit_low   = limit & 0xFFFF;
    gdt[index].granularity  = ((limit >> 16) & 0x0F) | (gran & 0xF0);
    gdt[index].access       = access;
}

// External assembly function
extern void gdt_flush(uint64_t gdt_ptr);

void gdt_init(void) {
    // Null descriptor
    gdt_set_entry(0, 0, 0, 0, 0);

    // Kernel Code Segment (64-bit, ring 0)
    gdt_set_entry(1, 0, 0xFFFFF,
        0x9A,       // Present, DPL=0, Code, Exec+Read
        0xA0);      // Long mode, 4KB granularity

    // Kernel Data Segment (64-bit, ring 0)
    gdt_set_entry(2, 0, 0xFFFFF,
        0x92,       // Present, DPL=0, Data, Read+Write
        0xC0);      // 4KB granularity

    // User Code Segment (64-bit, ring 3)
    gdt_set_entry(3, 0, 0xFFFFF,
        0xFA,       // Present, DPL=3, Code, Exec+Read
        0xA0);      // Long mode

    // User Data Segment (64-bit, ring 3)
    gdt_set_entry(4, 0, 0xFFFFF,
        0xF2,       // Present, DPL=3, Data, Read+Write
        0xC0);

    // TSS Descriptor (occupies entries 5 and 6)
    uint64_t tss_addr = (uint64_t)&tss;
    uint32_t tss_limit = sizeof(TSS) - 1;

    // Clear TSS
    for (size_t i = 0; i < sizeof(TSS); i++)
        ((uint8_t*)&tss)[i] = 0;

    tss.iopb_offset = sizeof(TSS);

    // Set TSS descriptor (16 bytes)
    TSSDescriptor *tss_desc = (TSSDescriptor*)&gdt[5];
    tss_desc->limit_low   = tss_limit & 0xFFFF;
    tss_desc->base_low    = tss_addr & 0xFFFF;
    tss_desc->base_middle = (tss_addr >> 16) & 0xFF;
    tss_desc->access      = 0x89;   // Present, TSS (Available)
    tss_desc->granularity = (tss_limit >> 16) & 0x0F;
    tss_desc->base_high   = (tss_addr >> 24) & 0xFF;
    tss_desc->base_upper  = (tss_addr >> 32) & 0xFFFFFFFF;
    tss_desc->reserved    = 0;

    // Load GDT
    gdt_ptr.limit = sizeof(gdt) - 1;
    gdt_ptr.base  = (uint64_t)&gdt;

    gdt_flush((uint64_t)&gdt_ptr);

    // Load TSS
    __asm__ volatile("ltr %%ax" : : "a"(0x28));
}

void tss_set_kernel_stack(uint64_t stack) {
    tss.rsp0 = stack;
}

void tss_init(void) {
    // Set default kernel stack
    extern uint8_t stack_top[];
    tss.rsp0 = (uint64_t)stack_top;

    // Set IST entries for special interrupt stacks

static void idt_set_gate(uint8_t num, uint64_t handler,
                         uint16_t selector, uint8_t ist,
                         uint8_t type_attr) {
    idt[num].offset_low  = handler & 0xFFFF;
    idt[num].offset_mid  = (handler >> 16) & 0xFFFF;
    idt[num].offset_high = (handler >> 32) & 0xFFFFFFFF;
    idt[num].selector    = selector;
    idt[num].ist         = ist & 0x07;
    idt[num].type_attr   = type_attr;
    idt[num].reserved    = 0;

// Interrupt type attributes
#define IDT_INTERRUPT_GATE  0x8E    // Present, DPL=0, 64-bit interrupt gate
#define IDT_TRAP_GATE       0x8F    // Present, DPL=0, 64-bit trap gate
#define IDT_USER_INT_GATE   0xEE    // Present, DPL=3, 64-bit interrupt gate

void idt_init(void) {
    // Clear IDT
    for (int i = 0; i < IDT_ENTRIES; i++) {
        idt_set_gate(i, 0, 0, 0, 0);
    }

    // ========== CPU Exceptions (0-31) ==========
    idt_set_gate(0,  (uint64_t)isr0,  0x08, 0, IDT_TRAP_GATE);     // Divide by zero
    idt_set_gate(1,  (uint64_t)isr1,  0x08, 0, IDT_TRAP_GATE);     // Debug
    idt_set_gate(2,  (uint64_t)isr2,  0x08, 2, IDT_INTERRUPT_GATE); // NMI (IST2)
    idt_set_gate(3,  (uint64_t)isr3,  0x08, 0, IDT_TRAP_GATE);     // Breakpoint
    idt_set_gate(4,  (uint64_t)isr4,  0x08, 0, IDT_TRAP_GATE);     // Overflow
    idt_set_gate(5,  (uint64_t)isr5,  0x08, 0, IDT_TRAP_GATE);     // Bound range
    idt_set_gate(6,  (uint64_t)isr6,  0x08, 0, IDT_TRAP_GATE);     // Invalid opcode
    idt_set_gate(7,  (uint64_t)isr7,  0x08, 0, IDT_TRAP_GATE);     // Device N/A
    idt_set_gate(8,  (uint64_t)isr8,  0x08, 1, IDT_INTERRUPT_GATE); // Double fault (IST1)
    idt_set_gate(9,  (uint64_t)isr9,  0x08, 0, IDT_TRAP_GATE);     // Coprocessor
    idt_set_gate(10, (uint64_t)isr10, 0x08, 0, IDT_TRAP_GATE);     // Invalid TSS
    idt_set_gate(11, (uint64_t)isr11, 0x08, 0, IDT_TRAP_GATE);     // Segment N/P
    idt_set_gate(12, (uint64_t)isr12, 0x08, 0, IDT_TRAP_GATE);     // Stack-segment
    idt_set_gate(13, (uint64_t)isr13, 0x08, 0, IDT_TRAP_GATE);     // General protection
    idt_set_gate(14, (uint64_t)isr14, 0x08, 3, IDT_INTERRUPT_GATE); // Page fault (IST3)
    // 15 reserved
    idt_set_gate(16, (uint64_t)isr16, 0x08, 0, IDT_TRAP_GATE);     // x87 FP
    idt_set_gate(17, (uint64_t)isr17, 0x08, 0, IDT_TRAP_GATE);     // Alignment
    idt_set_gate(18, (uint64_t)isr18, 0x08, 0, IDT_TRAP_GATE);     // Machine check
    idt_set_gate(19, (uint64_t)isr19, 0x08, 0, IDT_TRAP_GATE);     // SIMD FP
    idt_set_gate(20, (uint64_t)isr20, 0x08, 0, IDT_TRAP_GATE);     // Virtualization

    pic_remap();

    idt_set_gate(38, (uint64_t)irq6,  0x08, 0, IDT_INTERRUPT_GATE); // Floppy
    idt_set_gate(39, (uint64_t)irq7,  0x08, 0, IDT_INTERRUPT_GATE); // LPT1
    idt_set_gate(40, (uint64_t)irq8,  0x08, 0, IDT_INTERRUPT_GATE); // CMOS RTC
    idt_set_gate(41, (uint64_t)irq9,  0x08, 0, IDT_INTERRUPT_GATE);
    idt_set_gate(42, (uint64_t)irq10, 0x08, 0, IDT_INTERRUPT_GATE);
    idt_set_gate(43, (uint64_t)irq11, 0x08, 0, IDT_INTERRUPT_GATE);
    idt_set_gate(44, (uint64_t)irq12, 0x08, 0, IDT_INTERRUPT_GATE); // PS/2 Mouse

static inline void outb(uint16_t port, uint8_t val) {
    __asm__ volatile("outb %0, %1" : : "a"(val), "Nd"(port));
}

static inline uint8_t inb(uint16_t port) {
    uint8_t ret;
    __asm__ volatile("inb %1, %0" : "=a"(ret) : "Nd"(port));
    return ret;
}

static inline void io_wait(void) {
    outb(0x80, 0);
}

void pic_remap(void) {
    // Save masks
    uint8_t mask1 = inb(0x21);
    uint8_t mask2 = inb(0xA1);

    // Start initialization
    outb(0x20, 0x11); io_wait();
    outb(0xA0, 0x11); io_wait();

    // Set offsets
    outb(0x21, 0x20); io_wait();  // Master offset = 32
    outb(0xA1, 0x28); io_wait();  // Slave offset = 40

    // Configure cascade
    outb(0x21, 0x04); io_wait();  // Slave on IRQ2
    outb(0xA1, 0x02); io_wait();  // Cascade identity

    // 8086 mode
    outb(0x21, 0x01); io_wait();
    outb(0xA1, 0x01); io_wait();

    // Restore masks
    outb(0x21, mask1);
    outb(0xA1, mask2);
}

// Main interrupt dispatcher (called from assembly)
void isr_handler(InterruptFrame *frame) {
    uint64_t int_no = frame->int_no;

    if (int_no < 32) {
        // CPU Exception
        extern void kprintf(const char *fmt, ...);
        kprintf("\n[KERNEL] EXCEPTION: %s (INT %lu)\n",
                int_no < 21 ? exception_messages[int_no] : "Unknown",
                int_no);
        kprintf("[KERNEL] Error Code: 0x%lx\n", frame->error_code);
        kprintf("[KERNEL] RIP: 0x%lx  RSP: 0x%lx\n", frame->rip, frame->rsp);
        kprintf("[KERNEL] RAX: 0x%lx  RBX: 0x%lx  RCX: 0x%lx\n",
                frame->rax, frame->rbx, frame->rcx);

        if (int_no == 14) {
            // Page fault — get CR2 (faulting address)
            uint64_t cr2;
            __asm__ volatile("mov %%cr2, %0" : "=r"(cr2));
            kprintf("[KERNEL] Page Fault Address: 0x%lx\n", cr2);
            kprintf("[KERNEL] Flags: %s %s %s\n",
                    frame->error_code & 1 ? "Protection" : "Not-Present",
                    frame->error_code & 2 ? "Write" : "Read",
                    frame->error_code & 4 ? "User" : "Kernel");

            // Try to handle via page fault handler
            extern int page_fault_handler(uint64_t addr, uint64_t error, InterruptFrame *f);
            if (page_fault_handler(cr2, frame->error_code, frame) == 0) {
                return; // Handled
            }
        }

        // Unhandled exception — halt if in kernel, kill process if in user
        if (frame->cs == 0x08) {
            // Kernel panic
            kprintf("[KERNEL] *** KERNEL PANIC ***\n");
            __asm__ volatile("cli; hlt");
        } else {
            // Kill user process
            extern void process_kill_current(int signal);
            process_kill_current(11); // SIGSEGV
        }

    } else if (int_no >= 32 && int_no < 48) {
        // Hardware IRQ
        uint8_t irq = int_no - 32;

        if (irq_handlers[irq]) {
            irq_handlers[irq](frame);
        }

        // Send EOI
        if (irq >= 8) {
            outb(0xA0, 0x20);  // Slave PIC EOI
        }
        outb(0x20, 0x20);      // Master PIC EOI

    } else if (int_no == 128) {
        // System call (int 0x80)
        extern void syscall_handler(InterruptFrame *frame);
        syscall_handler(frame);
    }
}

5.1 Physical Memory Manager (Bitmap-based)

C

// kernel/mm/pmm.c
// Physical Memory Manager using bitmap allocation

#include <devil/types.h>
#include "pmm.h"

#define BITMAP_SIZE     (1024 * 1024)   // Support up to 32GB (1M bits * 32KB pages... actually 4KB)
// 1M bits = 128KB bitmap = 4GB of physical memory tracked

static uint64_t bitmap[BITMAP_SIZE / 64]; // Each uint64_t tracks 64 pages
static uint64_t total_memory = 0;
static uint64_t used_pages = 0;
static uint64_t total_pages = 0;

// E820 memory map entry
typedef struct {
    uint64_t base;
    uint64_t length;
    uint32_t type;      // 1 = usable, 2 = reserved, 3 = ACPI reclaimable
    uint32_t acpi_attr;
} PACKED E820Entry;

static inline void bitmap_set(uint64_t page) {
    bitmap[page / 64] |= (1ULL << (page % 64));
}

static inline void bitmap_clear(uint64_t page) {
    bitmap[page / 64] &= ~(1ULL << (page % 64));
}

static inline bool bitmap_test(uint64_t page) {
    return (bitmap[page / 64] >> (page % 64)) & 1;
}

void pmm_init(uint64_t memory_map_addr, uint64_t entry_count) {
    extern void kprintf(const char *fmt, ...);

    // Mark all memory as used initially
    for (size_t i = 0; i < BITMAP_SIZE / 64; i++) {
        bitmap[i] = 0xFFFFFFFFFFFFFFFF;
    }

    // Parse E820 memory map
    E820Entry *entries = (E820Entry*)memory_map_addr;

    for (uint64_t i = 0; i < entry_count; i++) {
        kprintf("[PMM] Region: base=0x%lx, length=0x%lx, type=%u\n",
                entries[i].base, entries[i].length, entries[i].type);

        if (entries[i].type == 1) { // Usable RAM
            uint64_t base = ALIGN_UP(entries[i].base, PAGE_SIZE);
            uint64_t end  = ALIGN_DOWN(entries[i].base + entries[i].length, PAGE_SIZE);

            for (uint64_t addr = base; addr < end; addr += PAGE_SIZE) {
                uint64_t page = addr / PAGE_SIZE;
                if (page < BITMAP_SIZE) {
                    bitmap_clear(page);
                    total_pages++;
                }
            }

            total_memory += entries[i].length;
        }
    }

    // Reserve first 2MB (bootloader, kernel, etc.)
    for (uint64_t page = 0; page < 512; page++) {
        bitmap_set(page);
        used_pages++;
    }

    // Reserve kernel area (assume kernel at 1MB, size 4MB)
    for (uint64_t page = 256; page < 1280; page++) {
        if (!bitmap_test(page)) {
            bitmap_set(page);
            used_pages++;
        }
    }

    kprintf("[PMM] Total memory: %lu MB\n", total_memory / (1024 * 1024));
    kprintf("[PMM] Total pages: %lu (%lu MB usable)\n",
            total_pages, (total_pages * PAGE_SIZE) / (1024 * 1024));
    kprintf("[PMM] Reserved pages: %lu\n", used_pages);
}

// Allocate a single physical page
void* pmm_alloc_page(void) {
    for (size_t i = 0; i < BITMAP_SIZE / 64; i++) {
        if (bitmap[i] != 0xFFFFFFFFFFFFFFFF) {
            // Found a free bit
            for (int bit = 0; bit < 64; bit++) {
                if (!(bitmap[i] & (1ULL << bit))) {
                    uint64_t page = i * 64 + bit;
                    bitmap_set(page);
                    used_pages++;
                    return (void*)(page * PAGE_SIZE);
                }
            }
        }
    }
    return NULL; // Out of memory
}

// Allocate contiguous pages
void* pmm_alloc_pages(size_t count) {
    size_t found = 0;
    size_t start_page = 0;

    for (size_t page = 0; page < total_pages; page++) {
        if (!bitmap_test(page)) {
            if (found == 0) start_page = page;
            found++;
            if (found == count) {
                // Allocate all pages
                for (size_t j = start_page; j < start_page + count; j++) {
                    bitmap_set(j);
                    used_pages++;
                }
                return (void*)(start_page * PAGE_SIZE);
            }
        } else {
            found = 0;
        }
    }
    return NULL;
}

// Free a physical page
void pmm_free_page(void *addr) {
    uint64_t page = (uint64_t)addr / PAGE_SIZE;
    if (page < BITMAP_SIZE && bitmap_test(page)) {
        bitmap_clear(page);
        used_pages--;
    }
}

void pmm_free_pages(void *addr, size_t count) {
    uint64_t page = (uint64_t)addr / PAGE_SIZE;
    for (size_t i = 0; i < count; i++) {
        if (page + i < BITMAP_SIZE && bitmap_test(page + i)) {
            bitmap_clear(page + i);
            used_pages--;
        }
    }
}

uint64_t pmm_get_free_pages(void) {
    return total_pages - used_pages;
}

uint64_t pmm_get_total_memory(void) {
    return total_memory;
}

// kernel/mm/vmm.c
// Virtual Memory Manager - Page table management for x86_64

#include <devil/types.h>
#include "vmm.h"
#include "pmm.h"

// Page table entry flags
#define PTE_PRESENT     (1ULL << 0)
#define PTE_WRITABLE    (1ULL << 1)
#define PTE_USER        (1ULL << 2)
#define PTE_WRITETHROUGH (1ULL << 3)
#define PTE_NOCACHE     (1ULL << 4)
#define PTE_ACCESSED    (1ULL << 5)
#define PTE_DIRTY       (1ULL << 6)
#define PTE_HUGE        (1ULL << 7)     // 2MB page
#define PTE_GLOBAL      (1ULL << 8)
#define PTE_NX          (1ULL << 63)    // No Execute

#define PTE_ADDR_MASK   0x000FFFFFFFFFF000ULL

// Page table structure (4-level: PML4 -> PDPT -> PD -> PT)
typedef uint64_t page_table_t[512] ALIGNED(PAGE_SIZE);

// Current PML4
static uint64_t *kernel_pml4 = NULL;

// Physical to virtual address conversion (identity mapped in kernel)
static inline uint64_t phys_to_virt(uint64_t phys) {
    return phys; // For now, identity mapped. Later: phys + KERNEL_VIRTUAL_BASE
}

static inline uint64_t virt_to_phys(uint64_t virt) {
    return virt; // For now, identity mapped
}

// Get or create page table entry at a given level
static uint64_t* vmm_get_or_create_table(uint64_t *table, uint64_t index, uint64_t flags) {
    if (!(table[index] & PTE_PRESENT)) {
        // Allocate new page table
        void *new_table = pmm_alloc_page();
        if (!new_table) return NULL;

        // Zero it out
        uint64_t *nt = (uint64_t*)phys_to_virt((uint64_t)new_table);
        for (int i = 0; i < 512; i++) nt[i] = 0;

        table[index] = (uint64_t)new_table | flags | PTE_PRESENT;
    }

    return (uint64_t*)phys_to_virt(table[index] & PTE_ADDR_MASK);
}

void vmm_init(void) {
    extern void kprintf(const char *fmt, ...);

    // Allocate new PML4 for kernel
    kernel_pml4 = (uint64_t*)pmm_alloc_page();
    if (!kernel_pml4) {
        kprintf("[VMM] FATAL: Cannot allocate PML4!\n");
        return;
    }

    // Clear PML4
    for (int i = 0; i < 512; i++) kernel_pml4[i] = 0;

    // Identity map first 4GB (for kernel access to physical memory)
    // Using 2MB pages for efficiency
    for (uint64_t addr = 0; addr < 0x100000000ULL; addr += 0x200000) {
        vmm_map_page_2mb(kernel_pml4, addr, addr,
                         PTE_PRESENT | PTE_WRITABLE | PTE_GLOBAL);
    }

    // Load new page tables
    __asm__ volatile("mov %0, %%cr3" : : "r"((uint64_t)kernel_pml4));

    kprintf("[VMM] Page tables initialized. 4GB identity mapped.\n");
}

// Map a 4KB page
int vmm_map_page(uint64_t *pml4, uint64_t virt, uint64_t phys, uint64_t flags) {
    uint64_t pml4_idx = (virt >> 39) & 0x1FF;
    uint64_t pdpt_idx = (virt >> 30) & 0x1FF;
    uint64_t pd_idx   = (virt >> 21) & 0x1FF;
    uint64_t pt_idx   = (virt >> 12) & 0x1FF;

    uint64_t *pdpt = vmm_get_or_create_table(pml4, pml4_idx,
                                              PTE_WRITABLE | PTE_USER);
    if (!pdpt) return -1;

    uint64_t *pd = vmm_get_or_create_table(pdpt, pdpt_idx,
                                            PTE_WRITABLE | PTE_USER);
    if (!pd) return -1;

    uint64_t *pt = vmm_get_or_create_table(pd, pd_idx,
                                            PTE_WRITABLE | PTE_USER);
    if (!pt) return -1;

    pt[pt_idx] = (phys & PTE_ADDR_MASK) | flags | PTE_PRESENT;

    // Invalidate TLB for this address
    __asm__ volatile("invlpg (%0)" : : "r"(virt) : "memory");

    return 0;
}

// Map a 2MB page
int vmm_map_page_2mb(uint64_t *pml4, uint64_t virt, uint64_t phys, uint64_t flags) {
    uint64_t pml4_idx = (virt >> 39) & 0x1FF;
    uint64_t pdpt_idx = (virt >> 30) & 0x1FF;
    uint64_t pd_idx   = (virt >> 21) & 0x1FF;

    uint64_t *pdpt = vmm_get_or_create_table(pml4, pml4_idx,
                                              PTE_WRITABLE);
    if (!pdpt) return -1;

    uint64_t *pd = vmm_get_or_create_table(pdpt, pdpt_idx,
                                            PTE_WRITABLE);
    if (!pd) return -1;

    pd[pd_idx] = (phys & 0xFFFFFFFFFFE00000ULL) | flags | PTE_PRESENT | PTE_HUGE;

    __asm__ volatile("invlpg (%0)" : : "r"(virt) : "memory");
    return 0;
}

// Unmap a page
void vmm_unmap_page(uint64_t *pml4, uint64_t virt) {
    uint64_t pml4_idx = (virt >> 39) & 0x1FF;
    uint64_t pdpt_idx = (virt >> 30) & 0x1FF;
    uint64_t pd_idx   = (virt >> 21) & 0x1FF;
    uint64_t pt_idx   = (virt >> 12) & 0x1FF;

    if (!(pml4[pml4_idx] & PTE_PRESENT)) return;
    uint64_t *pdpt = (uint64_t*)phys_to_virt(pml4[pml4_idx] & PTE_ADDR_MASK);

    if (!(pdpt[pdpt_idx] & PTE_PRESENT)) return;
    uint64_t *pd = (uint64_t*)phys_to_virt(pdpt[pdpt_idx] & PTE_ADDR_MASK);

    if (!(pd[pd_idx] & PTE_PRESENT)) return;
    if (pd[pd_idx] & PTE_HUGE) {
        pd[pd_idx] = 0;
    } else {
        uint64_t *pt = (uint64_t*)phys_to_virt(pd[pd_idx] & PTE_ADDR_MASK);
        pt[pt_idx] = 0;
    }

    __asm__ volatile("invlpg (%0)" : : "r"(virt) : "memory");
}

// Create a new address space (for user processes)
uint64_t* vmm_create_address_space(void) {
    uint64_t *new_pml4 = (uint64_t*)phys_to_virt((uint64_t)pmm_alloc_page());
    if (!new_pml4) return NULL;

    // Clear user space entries
    for (int i = 0; i < 256; i++) new_pml4[i] = 0;

    // Copy kernel space entries (upper half: entries 256-511)
    for (int i = 256; i < 512; i++) {
        new_pml4[i] = kernel_pml4[i];
    }

    return new_pml4;
}

// Switch address space
void vmm_switch_address_space(uint64_t *pml4) {
    uint64_t phys = virt_to_phys((uint64_t)pml4);
    __asm__ volatile("mov %0, %%cr3" : : "r"(phys) : "memory");
}

uint64_t* vmm_get_kernel_pml4(void) {
    return kernel_pml4;
}

// Get physical address for a virtual address
uint64_t vmm_get_physical(uint64_t *pml4, uint64_t virt) {
    uint64_t pml4_idx = (virt >> 39) & 0x1FF;
    uint64_t pdpt_idx = (virt >> 30) & 0x1FF;
    uint64_t pd_idx   = (virt >> 21) & 0x1FF;
    uint64_t pt_idx   = (virt >> 12) & 0x1FF;

    if (!(pml4[pml4_idx] & PTE_PRESENT)) return 0;
    uint64_t *pdpt = (uint64_t*)phys_to_virt(pml4[pml4_idx] & PTE_ADDR_MASK);

    if (!(pdpt[pdpt_idx] & PTE_PRESENT)) return 0;
    uint64_t *pd = (uint64_t*)phys_to_virt(pdpt[pdpt_idx] & PTE_ADDR_MASK);

    if (!(pd[pd_idx] & PTE_PRESENT)) return 0;
    if (pd[pd_idx] & PTE_HUGE) {
        return (pd[pd_idx] & 0xFFFFFFFFFFE00000ULL) + (virt & 0x1FFFFF);
    }

    uint64_t *pt = (uint64_t*)phys_to_virt(pd[pd_idx] & PTE_ADDR_MASK);
    if (!(pt[pt_idx] & PTE_PRESENT)) return 0;

    return (pt[pt_idx] & PTE_ADDR_MASK) + (virt & 0xFFF);
}

5.3 Kernel Heap (Slab Allocator)

C

// kernel/mm/heap.c
// Kernel heap using slab allocation

#include <devil/types.h>
#include "heap.h"
#include "pmm.h"
#include "vmm.h"

// ============================================
// Simple block allocator for kernel malloc/free
// Uses a free list with block headers
// ============================================

typedef struct heap_block {
    size_t size;
    bool free;
    struct heap_block *next;
    struct heap_block *prev;
    uint32_t magic;         // For corruption detection: 0xDEADBEEF
} HeapBlock;

#define HEAP_MAGIC      0xDEADBEEF
#define HEAP_HEADER_SIZE sizeof(HeapBlock)
#define MIN_BLOCK_SIZE  32

static HeapBlock *heap_start = NULL;
static HeapBlock *heap_end = NULL;
static uint64_t heap_base = 0;
static uint64_t heap_size = 0;
static uint64_t heap_used = 0;

// Spinlock for thread safety
static volatile uint64_t heap_lock = 0;

static inline void spin_lock(volatile uint64_t *lock) {
    while (__sync_lock_test_and_set(lock, 1)) {
        __asm__ volatile("pause");
    }
}

static inline void spin_unlock(volatile uint64_t *lock) {
    __sync_lock_release(lock);
}

void heap_init(void) {
    extern void kprintf(const char *fmt, ...);

    // Allocate initial heap pages (16MB)
    size_t initial_pages = KERNEL_HEAP_INITIAL / PAGE_SIZE;
    void *heap_phys = pmm_alloc_pages(initial_pages);
    if (!heap_phys) {
        kprintf("[HEAP] FATAL: Cannot allocate heap memory!\n");
        return;
    }

    // Map heap to virtual address
    heap_base = KERNEL_HEAP_START;
    for (size_t i = 0; i < initial_pages; i++) {
        vmm_map_page(vmm_get_kernel_pml4(),
                     heap_base + i * PAGE_SIZE,
                     (uint64_t)heap_phys + i * PAGE_SIZE,
                     PTE_PRESENT | PTE_WRITABLE);
    }

    heap_size = KERNEL_HEAP_INITIAL;

    // Initialize first block (entire heap as one free block)
    heap_start = (HeapBlock*)heap_base;
    heap_start->size = heap_size - HEAP_HEADER_SIZE;
    heap_start->free = true;
    heap_start->next = NULL;
    heap_start->prev = NULL;
    heap_start->magic = HEAP_MAGIC;
    heap_end = heap_start;

    kprintf("[HEAP] Initialized: base=0x%lx, size=%lu MB\n",
            heap_base, heap_size / (1024 * 1024));
}

// Find best-fit free block
static HeapBlock* find_best_fit(size_t size) {
    HeapBlock *best = NULL;
    HeapBlock *current = heap_start;

    while (current) {
        if (current->free && current->size >= size) {
            if (!best || current->size < best->size) {
                best = current;
                if (best->size == size) break; // Perfect fit
            }
        }
        current = current->next;
    }
    return best;
}

// Split a block if possible
static void split_block(HeapBlock *block, size_t size) {
    if (block->size > size + HEAP_HEADER_SIZE + MIN_BLOCK_SIZE) {
        HeapBlock *new_block = (HeapBlock*)((uint8_t*)block + HEAP_HEADER_SIZE + size);
        new_block->size = block->size - size - HEAP_HEADER_SIZE;
        new_block->free = true;
        new_block->next = block->next;
        new_block->prev = block;
        new_block->magic = HEAP_MAGIC;

        if (block->next) block->next->prev = new_block;
        block->next = new_block;
        block->size = size;

        if (block == heap_end) heap_end = new_block;
    }
}

// Merge adjacent free blocks
static void merge_free_blocks(HeapBlock *block) {
    // Merge with next
    if (block->next && block->next->free) {
        block->size += HEAP_HEADER_SIZE + block->next->size;
        block->next = block->next->next;
        if (block->next) block->next->prev = block;
        if (block->next == NULL) heap_end = block;
    }

    // Merge with previous
    if (block->prev && block->prev->free) {
        block->prev->size += HEAP_HEADER_SIZE + block->size;
        block->prev->next = block->next;
        if (block->next) block->next->prev = block->prev;
        if (block == heap_end) heap_end = block->prev;
    }
}

void* kmalloc(size_t size) {
    if (size == 0) return NULL;

    // Align size to 16 bytes
    size = ALIGN_UP(size, 16);

    spin_lock(&heap_lock);

    HeapBlock *block = find_best_fit(size);
    if (!block) {
        // TODO: Expand heap
        spin_unlock(&heap_lock);
        return NULL;
    }

    split_block(block, size);
    block->free = false;
    heap_used += block->size + HEAP_HEADER_SIZE;

    spin_unlock(&heap_lock);

    // Return pointer after header
    return (void*)((uint8_t*)block + HEAP_HEADER_SIZE);
}

void* kmalloc_aligned(size_t size, size_t alignment) {
    size_t total = size + alignment + HEAP_HEADER_SIZE;
    void *ptr = kmalloc(total);
    if (!ptr) return NULL;

    uintptr_t aligned = ALIGN_UP((uintptr_t)ptr, alignment);
    return (void*)aligned;
}

void* kzalloc(size_t size) {
    void *ptr = kmalloc(size);
    if (ptr) {
        uint8_t *p = (uint8_t*)ptr;
        for (size_t i = 0; i < size; i++) p[i] = 0;
    }
    return ptr;
}

void kfree(void *ptr) {
    if (!ptr) return;

    spin_lock(&heap_lock);

    HeapBlock *block = (HeapBlock*)((uint8_t*)ptr - HEAP_HEADER_SIZE);

    // Validate magic
    if (block->magic != HEAP_MAGIC) {
        extern void kprintf(const char *fmt, ...);
        kprintf("[HEAP] WARNING: Corrupt heap block at 0x%lx!\n", (uint64_t)ptr);
        spin_unlock(&heap_lock);
        return;
    }

    block->free = true;
    heap_used -= block->size + HEAP_HEADER_SIZE;

    merge_free_blocks(block);

    spin_unlock(&heap_lock);
}

void* krealloc(void *ptr, size_t new_size) {
    if (!ptr) return kmalloc(new_size);
    if (new_size == 0) {
        kfree(ptr);
        return NULL;
    }

    HeapBlock *block = (HeapBlock*)((uint8_t*)ptr - HEAP_HEADER_SIZE);
    if (block->size >= new_size) return ptr;

    void *new_ptr = kmalloc(new_size);
    if (new_ptr) {
        uint8_t *src = (uint8_t*)ptr;
        uint8_t *dst = (uint8_t*)new_ptr;
        size_t copy_size = MIN(block->size, new_size);
        for (size_t i = 0; i < copy_size; i++) dst[i] = src[i];
        kfree(ptr);
    }
    return new_ptr;
}

size_t heap_get_used(void) { return heap_used; }
size_t heap_get_free(void) { return heap_size - heap_used; }

// kernel/proc/process.c

#include "process.h"
#include "../mm/pmm.h"
#include "../mm/vmm.h"
#include "../mm/heap.h"
#include "../fs/vfs.h"

static Process *process_table[MAX_PROCESSES] = {0};
static pid_t next_pid = 1;
static Process *current_process = NULL;

// Spinlock
static volatile uint64_t proc_lock = 0;

static pid_t allocate_pid(void) {
    for (pid_t i = next_pid; i < MAX_PROCESSES; i++) {
        if (!process_table[i]) {
            next_pid = i + 1;
            return i;
        }
    }
    // Wrap around
    for (pid_t i = 1; i < next_pid; i++) {
        if (!process_table[i]) {
            next_pid = i + 1;
            return i;
        }
    }
    return -1; // No free PIDs
}

void process_init(void) {
    extern void kprintf(const char *fmt, ...);

    // Create kernel process (PID 0)
    Process *kernel_proc = (Process*)kzalloc(sizeof(Process));
    kernel_proc->pid = 0;
    kernel_proc->ppid = 0;

    // Copy name
    const char *name = "[kernel]";
    for (int i = 0; name[i] && i < 255; i++) kernel_proc->name[i] = name[i];

    kernel_proc->state = PROCESS_STATE_RUNNING;
    kernel_proc->page_table = vmm_get_kernel_pml4();
    kernel_proc->uid = 0;
    kernel_proc->gid = 0;
    kernel_proc->priority = 0;

    // Set cwd to "/"
    kernel_proc->cwd[0] = '/';
    kernel_proc->cwd[1] = '\0';

    process_table[0] = kernel_proc;
    current_process = kernel_proc;

    kprintf("[PROC] Process subsystem initialized.\n");
}

Process* process_create(const char *name, const char *path,
                        char **argv, char **envp) {
    extern void kprintf(const char *fmt, ...);

    Process *proc = (Process*)kzalloc(sizeof(Process));
    if (!proc) return NULL;

    // Allocate PID
    proc->pid = allocate_pid();
    if (proc->pid < 0) {
        kfree(proc);
        return NULL;
    }

    // Set name
    for (int i = 0; name[i] && i < 255; i++) proc->name[i] = name[i];

    proc->ppid = current_process ? current_process->pid : 0;
    proc->parent = current_process;
    proc->state = PROCESS_STATE_READY;

    // Create address space
    proc->page_table = vmm_create_address_space();
    if (!proc->page_table) {
        kfree(proc);
        return NULL;
    }

    // Allocate kernel stack (64KB)
    proc->kernel_stack_size = KERNEL_STACK_SIZE;
    void *kstack = pmm_alloc_pages(proc->kernel_stack_size / PAGE_SIZE);
    if (!kstack) {
        kfree(proc);
        return NULL;
    }
    proc->kernel_stack = (uint64_t)kstack + proc->kernel_stack_size;

    // Allocate user stack (8MB at top of user space)
    proc->user_stack_size = USER_STACK_SIZE;
    uint64_t user_stack_base = 0x00007FFFFFFFE000ULL - USER_STACK_SIZE;
    for (size_t i = 0; i < USER_STACK_SIZE / PAGE_SIZE; i++) {
        void *page = pmm_alloc_page();
        vmm_map_page(proc->page_table,
                     user_stack_base + i * PAGE_SIZE,
                     (uint64_t)page,
                     PTE_PRESENT | PTE_WRITABLE | PTE_USER);
    }
    proc->user_stack = 0x00007FFFFFFFE000ULL;

    // Load ELF binary
    extern uint64_t elf_load(uint64_t *pml4, const char *path);
    uint64_t entry_point = elf_load(proc->page_table, path);
    if (entry_point == 0) {
        kprintf("[PROC] Failed to load ELF: %s\n", path);
        kfree(proc);
        return NULL;
    }

    // Set up initial context
    proc->context.rip = entry_point;
    proc->context.rsp = proc->user_stack;
    proc->context.rflags = 0x202;       // IF set
    proc->context.cs = 0x18 | 3;        // User code segment, RPL=3
    proc->context.ss = 0x20 | 3;        // User data segment, RPL=3

    // Set up default file descriptors
    // stdin (0), stdout (1), stderr (2)
    extern FileDescriptor* tty_create_fd(void);
    proc->fd_table[0] = tty_create_fd(); // stdin
    proc->fd_table[1] = tty_create_fd(); // stdout
    proc->fd_table[2] = tty_create_fd(); // stderr

    // Set cwd
    if (current_process) {
        for (int i = 0; i < MAX_PATH; i++) {
            proc->cwd[i] = current_process->cwd[i];
            if (!proc->cwd[i]) break;
        }
    } else {
        proc->cwd[0] = '/';
        proc->cwd[1] = '\0';
    }

    // Set credentials (inherit from parent or default)
    if (current_process) {
        proc->uid = current_process->uid;
        proc->gid = current_process->gid;
        proc->euid = current_process->euid;
        proc->egid = current_process->egid;
    }

    // Default priority
    proc->priority = 20;
    proc->nice = 0;

    // Register in process table
    process_table[proc->pid] = proc;

    // Add to scheduler
    extern void scheduler_add_process(Process *proc);
    scheduler_add_process(proc);

    kprintf("[PROC] Created process '%s' (PID %d)\n", name, proc->pid);
    return proc;
}

int process_create_init(const char *path) {

        "HOME=/root",

        "SHELL=/bin/devil_shell",
        NULL
    };

}

    return current_process;
}

// kernel/proc/scheduler.c
// Completely Fair Scheduler (CFS) inspired

#include "process.h"
#include "scheduler.h"
#include <devil/types.h>

// Red-black tree for scheduling (simplified as a sorted list for now)
typedef struct run_queue {
    Process *head;
    Process *tail;
    uint32_t count;
    uint64_t min_vruntime;
} RunQueue;

static RunQueue run_queue = {0};
static volatile uint64_t sched_lock = 0;
static volatile uint64_t ticks = 0;
static uint64_t current_timeslice = 0;

// Timer constants
#define TICK_INTERVAL_NS    1000000     // 1ms per tick
#define MIN_TIMESLICE_NS    1000000     // 1ms minimum
#define DEFAULT_TIMESLICE   10          // 10ms (10 ticks)

// Weight table (based on nice values, like Linux CFS)
static const uint64_t nice_to_weight[] = {
    /* -20 */ 88761, 71755, 56483, 46273, 36291,
    /* -15 */ 29154, 23254, 18705, 14949, 11916,
    /* -10 */ 9548,  7620,  6100,  4904,  3906,
    /*  -5 */ 3121,  2501,  1991,  1586,  1277,
    /*   0 */ 1024,  820,   655,   526,   423,
    /*   5 */ 335,   272,   215,   172,   137,
    /*  10 */ 110,   87,    70,    56,    45,
    /*  15 */ 36,    29,    23,    18,    15,
};

static uint64_t get_weight(int nice) {
    int index = nice + 20;
    if (index < 0) index = 0;
    if (index > 39) index = 39;
    return nice_to_weight[index];
}

void scheduler_init(void) {
    extern void kprintf(const char *fmt, ...);

    run_queue.head = NULL;
    run_queue.tail = NULL;
    run_queue.count = 0;
    run_queue.min_vruntime = 0;

    // Register timer interrupt handler
    extern void irq_register_handler(uint8_t irq, void (*handler)(void*));
    irq_register_handler(0, (void*)scheduler_tick);

    kprintf("[SCHED] Scheduler initialized (CFS-like, %dms tick)\n",
            TICK_INTERVAL_NS / 1000000);
}

void scheduler_add_process(Process *proc) {
    spin_lock(&sched_lock);

    proc->vruntime = run_queue.min_vruntime;

    // Insert sorted by vruntime (leftmost = smallest vruntime = runs next)
    if (!run_queue.head) {
        run_queue.head = proc;
        run_queue.tail = proc;
        proc->next = NULL;
        proc->prev = NULL;
    } else {
        // Find insertion point
        Process *curr = run_queue.head;
        while (curr && curr->vruntime <= proc->vruntime) {
            curr = curr->next;
        }

        if (curr == run_queue.head) {
            // Insert at head
            proc->next = run_queue.head;
            proc->prev = NULL;
            run_queue.head->prev = proc;
            run_queue.head = proc;
        } else if (!curr) {
            // Insert at tail
            proc->prev = run_queue.tail;
            proc->next = NULL;
            run_queue.tail->next = proc;
            run_queue.tail = proc;
        } else {
            // Insert before curr
            proc->next = curr;
            proc->prev = curr->prev;
            curr->prev->next = proc;
            curr->prev = proc;
        }
    }

    run_queue.count++;
    spin_unlock(&sched_lock);
}

void scheduler_remove_process(Process *proc) {
    spin_lock(&sched_lock);

    if (proc->prev) proc->prev->next = proc->next;
    else run_queue.head = proc->next;

    if (proc->next) proc->next->prev = proc->prev;
    else run_queue.tail = proc->prev;

    proc->next = NULL;
    proc->prev = NULL;
    run_queue.count--;

    spin_unlock(&sched_lock);
}

// Called on every timer tick
void scheduler_tick(void *frame) {
    ticks++;

    Process *current = process_get_current();
    if (!current || current->pid == 0) return;

    // Update vruntime
    uint64_t weight = get_weight(current->nice);
    uint64_t delta = (TICK_INTERVAL_NS * 1024) / weight;
    current->vruntime += delta;
    current->cpu_time += TICK_INTERVAL_NS;

    // Check if timeslice expired
    current_timeslice++;
    if (current_timeslice >= DEFAULT_TIMESLICE) {
        schedule();
    }
}

// Main scheduling function
void schedule(void) {
    spin_lock(&sched_lock);

    Process *current = process_get_current();
    Process *next = NULL;

    if (current && current->state == PROCESS_STATE_RUNNING) {
        current->state = PROCESS_STATE_READY;
        // Re-insert into run queue
        scheduler_remove_process(current);
        scheduler_add_process(current);
    }

    // Pick process with smallest vruntime (head of queue)
    next = run_queue.head;
    while (next && next->state != PROCESS_STATE_READY) {
        next = next->next;
    }

    if (!next) {
        // No runnable process — run idle
        spin_unlock(&sched_lock);
        return;
    }

    next->state = PROCESS_STATE_RUNNING;
    current_timeslice = 0;

    // Update min_vruntime
    if (run_queue.head) {
        run_queue.min_vruntime = run_queue.head->vruntime;
    }

    spin_unlock(&sched_lock);

    if (next != current) {
        context_switch(current, next);
    }
}

// Context switch (most work done in assembly)
void context_switch(Process *old, Process *new) {
    // Save current process state
    // (Registers saved by ISR stub before calling scheduler_tick)

    // Switch address space
    vmm_switch_address_space(new->page_table);

    // Update TSS kernel stack
    extern void tss_set_kernel_stack(uint64_t stack);
    tss_set_kernel_stack(new->kernel_stack);

    // Set current process
    process_set_current(new);

    // Actual register context switch (in assembly)
    extern void switch_context(void *old_ctx, void *new_ctx);
    if (old) {
        switch_context(&old->context, &new->context);
    } else {
        switch_context(NULL, &new->context);
    }
}

    mov     [rdi + 0x08], rbx
    mov     [rdi + 0x10], rcx
    mov     [rdi + 0x18], rdx
    mov     [rdi + 0x20], rsi
    mov     [rdi + 0x28], rdi
    mov     [rdi + 0x30], rbp
    mov     [rdi + 0x38], rsp
    mov     [rdi + 0x40], r8
    mov     [rdi + 0x48], r9
    mov     [rdi + 0x50], r10
    mov     [rdi + 0x58], r11
    mov     [rdi + 0x60], r12
    mov     [rdi + 0x68], r13
    mov     [rdi + 0x70], r14
    mov     [rdi + 0x78], r15

    ; Save RIP (return address)
    lea     rax, [rel .restore_done]
    mov     [rdi + 0x80], rax

    ; Save RFLAGS
    pushfq
    pop     rax
    mov     [rdi + 0x88], rax

    ; Save FPU state
    lea     rax, [rdi + 0xA0]
    fxsave  [rax]

.restore:
    ; Restore new context
    ; Restore FPU state
    lea     rax, [rsi + 0xA0]
    fxrstor [rax]

    ; Restore general registers
    mov     rax, [rsi + 0x00]
    mov     rbx, [rsi + 0x08]
    mov     rcx, [rsi + 0x10]
    mov     rdx, [rsi + 0x18]
    ; rsi and rdi restored last
    mov     rbp, [rsi + 0x30]
    mov     rsp, [rsi + 0x38]
    mov     r8,  [rsi + 0x40]
    mov     r9,  [rsi + 0x48]
    mov     r10, [rsi + 0x50]
    mov     r11, [rsi + 0x58]
    mov     r12, [rsi + 0x60]
    mov     r13, [rsi + 0x68]
    mov     r14, [rsi + 0x70]
    mov     r15, [rsi + 0x78]

    ; Restore RFLAGS
    push    qword [rsi + 0x88]
    popfq

    ; Get RIP
    mov     rdi, [rsi + 0x28]      ; Restore RDI
    push    qword [rsi + 0x80]     ; Push RIP
    mov     rsi, [rsi + 0x20]      ; Restore RSI

    ret                             ; "Jump" to RIP

.restore_done:
    ret

// kernel/fs/vfs.c

#include "vfs.h"
#include "../mm/heap.h"

static VFSNode *root_node = NULL;
static FilesystemType *registered_fs = NULL;

// Mount points
typedef struct mount_point {
    char path[MAX_PATH];
    VFSNode *root;
    FilesystemType *fs;
    struct mount_point *next;
} MountPoint;

static MountPoint *mount_points = NULL;

void vfs_init(void) {
    extern void kprintf(const char *fmt, ...);

    // Create root directory node
    root_node = (VFSNode*)kzalloc(sizeof(VFSNode));
    root_node->name[0] = '/';
    root_node->type = VFS_DIRECTORY;
    root_node->permissions = S_IRUSR | S_IWUSR | S_IXUSR |
                             S_IRGRP | S_IXGRP |
                             S_IROTH | S_IXOTH;
    root_node->uid = 0;
    root_node->gid = 0;
    root_node->ref_count = 1;

    kprintf("[VFS] Virtual File System initialized.\n");
}

VFSNode* vfs_get_root(void) {
    return root_node;
}

// Path resolution
VFSNode* vfs_lookup(const char *path) {
    if (!path || path[0] != '/') return NULL;
    if (path[0] == '/' && path[1] == '\0') return root_node;

    VFSNode *current = root_node;
    char component[MAX_FILENAME + 1];
    int ci = 0;

    for (const char *p = path + 1; ; p++) {
        if (*p == '/' || *p == '\0') {
            component[ci] = '\0';
            if (ci > 0) {
                // Check for mount point
                MountPoint *mp = mount_points;
                while (mp) {
                    // Simple prefix match
                    // (Full implementation would be more sophisticated)
                    mp = mp->next;
                }

                // Look up in current directory
                VFSNode *found = NULL;
                if (current->ops && current->ops->finddir) {
                    found = current->ops->finddir(current, component);
                } else {
                    // Walk children list
                    VFSNode *child = current->children;
                    while (child) {
                        // strcmp
                        bool match = true;
                        for (int i = 0; ; i++) {
                            if (child->name[i] != component[i]) { match = false; break; }
                            if (child->name[i] == '\0') break;
                        }
                        if (match) { found = child; break; }
                        child = child->next_sibling;
                    }
                }

                if (!found) return NULL;

                // Follow mount points
                if (found->mounted_root) {
                    found = found->mounted_root;
                }

                current = found;
            }
            ci = 0;
            if (*p == '\0') break;
        } else {
            if (ci < MAX_FILENAME) {
                component[ci++] = *p;
            }
        }
    }

    return current;
}

void vfs_register_filesystem(FilesystemType *fs) {
    fs->next = registered_fs;
    registered_fs = fs;
    extern void kprintf(const char *fmt, ...);
    kprintf("[VFS] Registered filesystem: %s\n", fs->name);
}

int vfs_mount(const char *device, const char *mountpoint,
              const char *fstype, const char *options) {
    extern void kprintf(const char *fmt, ...);

    // Find filesystem type
    FilesystemType *fs = registered_fs;
    while (fs) {
        bool match = true;
        for (int i = 0; ; i++) {
            if (fs->name[i] != fstype[i]) { match = false; break; }
            if (fs->name[i] == '\0') break;
        }
        if (match) break;
        fs = fs->next;
    }

    if (!fs) {
        kprintf("[VFS] Unknown filesystem type: %s\n", fstype);
        return -1;
    }

    // Mount the filesystem
    VFSNode *fs_root = fs->mount(device, options);
    if (!fs_root) {
        kprintf("[VFS] Failed to mount %s on %s\n", fstype, mountpoint);
        return -1;
    }

    // Find mount point node
    VFSNode *mp_node = vfs_lookup(mountpoint);
    if (!mp_node) {
        // Create mount point directory
        vfs_mkdir(mountpoint, 0755);
        mp_node = vfs_lookup(mountpoint);
        if (!mp_node) return -1;
    }

    // Set mount
    mp_node->mounted_root = fs_root;
    mp_node->flags |= VFS_MOUNTPOINT;

    // Record mount point
    MountPoint *mp = (MountPoint*)kzalloc(sizeof(MountPoint));
    for (int i = 0; mountpoint[i] && i < MAX_PATH - 1; i++)
        mp->path[i] = mountpoint[i];
    mp->root = fs_root;
    mp->fs = fs;
    mp->next = mount_points;
    mount_points = mp;

    kprintf("[VFS] Mounted %s (%s) on %s\n", device ? device : "none", fstype, mountpoint);
    return 0;
}

// VFS operation wrappers
ssize_t vfs_read(VFSNode *node, uint64_t offset, size_t size, uint8_t *buffer) {
    if (!node || !node->ops || !node->ops->read) return -1;
    return node->ops->read(node, offset, size, buffer);
}

ssize_t vfs_write(VFSNode *node, uint64_t offset, size_t size, const uint8_t *buffer) {
    if (!node || !node->ops || !node->ops->write) return -1;
    return node->ops->write(node, offset, size, buffer);
}

int vfs_open(VFSNode *node, uint32_t flags) {
    if (!node) return -1;
    node->ref_count++;
    if (node->ops && node->ops->open)
        return node->ops->open(node, flags);
    return 0;
}

void vfs_close(VFSNode *node) {
    if (!node) return;
    node->ref_count--;
    if (node->ops && node->ops->close)
        node->ops->close(node);
}

dirent_t* vfs_readdir(VFSNode *node, uint32_t index) {
    if (!node || !(node->type & VFS_DIRECTORY)) return NULL;
    if (node->ops && node->ops->readdir)
        return node->ops->readdir(node, index);
    return NULL;
}

VFSNode* vfs_finddir(VFSNode *node, const char *name) {
    if (!node || !(node->type & VFS_DIRECTORY)) return NULL;
    if (node->ops && node->ops->finddir)
        return node->ops->finddir(node, name);
    return NULL;
}

int vfs_mkdir(const char *path, mode_t mode) {
    // Find parent directory
    char parent_path[MAX_PATH];
    char dir_name[MAX_FILENAME + 1];
    int last_slash = -1;

    for (int i = 0; path[i]; i++) {
        if (path[i] == '/') last_slash = i;
    }

    if (last_slash <= 0) {
        parent_path[0] = '/';
        parent_path[1] = '\0';
        int j = 0;
        for (int i = 1; path[i] && j < MAX_FILENAME; i++, j++)
            dir_name[j] = path[i];
        dir_name[j] = '\0';
    } else {
        for (int i = 0; i < last_slash && i < MAX_PATH - 1; i++)
            parent_path[i] = path[i];
        parent_path[last_slash] = '\0';
        int j = 0;
        for (int i = last_slash + 1; path[i] && j < MAX_FILENAME; i++, j++)
            dir_name[j] = path[i];
        dir_name[j] = '\0';
    }

    VFSNode *parent = vfs_lookup(parent_path);
    if (!parent) return -1;

    if (parent->ops && parent->ops->mkdir) {
        return parent->ops->mkdir(parent, dir_name, mode);
    }

    // Default: create in-memory directory node
    VFSNode *new_dir = (VFSNode*)kzalloc(sizeof(VFSNode));
    for (int i = 0; dir_name[i] && i < MAX_FILENAME; i++)
        new_dir->name[i] = dir_name[i];
    new_dir->type = VFS_DIRECTORY;
    new_dir->permissions = mode;
    new_dir->parent = parent;
    new_dir->ref_count = 1;

    // Add to parent's children
    new_dir->next_sibling = parent->children;
    parent->children = new_dir;

    return 0;
}

#ifndef _DEVILFS_H
#define _DEVILFS_H

#include <devil/types.h>

/*
 * DevilFS Layout:
 *
 * Block 0:       Superblock
 * Block 1-16:    Journal
 * Block 17-48:   Inode Bitmap
 * Block 49-80:   Block Bitmap
 * Block 81-N:    Inode Table
 * Block N+1-end: Data Blocks
 *
 * Features:
 * - 4KB block size
 * - Journaling (write-ahead log)
 * - Built-in encryption (AES-256)
 * - Maximum file size: 16TB (using extents)
 * - Maximum filesystem size: 256TB
 * - Maximum filename: 255 bytes
 * - POSIX permissions + extended attributes
 */

#define DEVILFS_MAGIC       0xDEV1C0DE
#define DEVILFS_VERSION     1
#define DEVILFS_BLOCK_SIZE  4096
#define DEVILFS_INODE_SIZE  256
#define DEVILFS_JOURNAL_BLOCKS 16

// Superblock
typedef struct {
    uint32_t magic;
    uint32_t version;
    uint64_t total_blocks;
    uint64_t total_inodes;
    uint64_t free_blocks;
    uint64_t free_inodes;
    uint32_t block_size;
    uint32_t inode_size;
    uint64_t first_data_block;
    uint64_t inode_table_block;
    uint64_t inode_bitmap_block;
    uint64_t block_bitmap_block;
    uint64_t journal_block;
    uint32_t journal_size;          // In blocks
    uint64_t root_inode;
    uint8_t  uuid[16];
    char     volume_name[64];
    uint32_t flags;
    uint8_t  encryption_key_hash[32]; // SHA-256 of encryption key
    uint32_t encryption_enabled;
    uint64_t mount_time;
    uint64_t write_time;
    uint32_t mount_count;

// Write a block to the device
static int devilfs_write_block(DevilFSInfo *fs, uint64_t block, const void *buffer) {
    uint64_t offset = block * DEVILFS_BLOCK_SIZE;

    // If encryption enabled, encrypt block before writing
    uint8_t encrypted[DEVILFS_BLOCK_SIZE];
    if (fs->encryption_enabled) {
        extern void aes256_encrypt(const uint8_t *key, const uint8_t *in,
                                   uint8_t *out, size_t len);
        aes256_encrypt(fs->encryption_key, (const uint8_t*)buffer,
                      encrypted, DEVILFS_BLOCK_SIZE);
        return vfs_write(fs->device, offset, DEVILFS_BLOCK_SIZE, encrypted);
    }

    return vfs_write(fs->device, offset, DEVILFS_BLOCK_SIZE, (const uint8_t*)buffer);
}

// Read an inode
static int devilfs_read_inode(DevilFSInfo *fs, uint64_t ino, DevilFSInode *inode) {
    uint64_t block = fs->sb.inode_table_block + (ino * DEVILFS_INODE_SIZE) / DEVILFS_BLOCK_SIZE;
    uint64_t offset = (ino * DEVILFS_INODE_SIZE) % DEVILFS_BLOCK_SIZE;

    uint8_t block_buf[DEVILFS_BLOCK_SIZE];
    devilfs_read_block(fs, block, block_buf);

    // Copy inode data
    uint8_t *src = block_buf + offset;
    uint8_t *dst = (uint8_t*)inode;
    for (size_t i = 0; i < sizeof(DevilFSInode); i++)
        dst[i] = src[i];

    return 0;
}

// Allocate a free inode
static uint64_t devilfs_alloc_inode(DevilFSInfo *fs) {
    for (uint64_t i = 0; i < fs->sb.total_inodes / 8; i++) {
        if (fs->inode_bitmap[i] != 0xFF) {
            for (int bit = 0; bit < 8; bit++) {
                if (!(fs->inode_bitmap[i] & (1 << bit))) {
                    fs->inode_bitmap[i] |= (1 << bit);
                    fs->sb.free_inodes--;
                    return i * 8 + bit;
                }
            }
        }
    }
    return 0; // No free inodes
}

// Allocate contiguous blocks
static uint64_t devilfs_alloc_blocks(DevilFSInfo *fs, uint32_t count) {
    uint32_t found = 0;
    uint64_t start = 0;

    for (uint64_t i = 0; i < fs->sb.total_blocks / 8; i++) {
        for (int bit = 0; bit < 8; bit++) {
            if (!(fs->block_bitmap[i] & (1 << bit))) {
                if (found == 0) start = i * 8 + bit;
                found++;
                if (found == count) {
                    // Mark blocks as used
                    for (uint32_t j = 0; j < count; j++) {
                        uint64_t b = start + j;
                        fs->block_bitmap[b / 8] |= (1 << (b % 8));
                    }
                    fs->sb.free_blocks -= count;
                    return start;
                }
            } else {
                found = 0;
            }
        }
    }
    return 0;
}

// File read implementation
static ssize_t devilfs_read(VFSNode *node, uint64_t offset, size_t size, uint8_t *buffer) {
    DevilFSInfo *fs = (DevilFSInfo*)node->fs_private;
    DevilFSInode inode;
    devilfs_read_inode(fs, node->inode, &inode);

    if (offset >= inode.size) return 0;
    if (offset + size > inode.size) size = inode.size - offset;

    size_t bytes_read = 0;
    uint8_t block_buf[DEVILFS_BLOCK_SIZE];

    while (bytes_read < size) {
        // Find which extent contains the current offset
        uint64_t file_block = (offset + bytes_read) / DEVILFS_BLOCK_SIZE;
        uint64_t block_offset = (offset + bytes_read) % DEVILFS_BLOCK_SIZE;

        // Search extents
        uint64_t disk_block = 0;
        uint64_t extent_offset = 0;
        for (int e = 0; e < 12; e++) {
            if (file_block < extent_offset + inode.extents[e].length) {
                disk_block = inode.extents[e].block + (file_block - extent_offset);
                break;
            }
            extent_offset += inode.extents[e].length;
        }

        if (disk_block == 0) break; // Past end of file

        devilfs_read_block(fs, disk_block, block_buf);

        size_t copy_size = DEVILFS_BLOCK_SIZE - block_offset;
        if (copy_size > size - bytes_read) copy_size = size - bytes_read;

        for (size_t i = 0; i < copy_size; i++)
            buffer[bytes_read + i] = block_buf[block_offset + i];

        bytes_read += copy_size;
    }

    return bytes_read;
}

// File write implementation
static ssize_t devilfs_write(VFSNode *node, uint64_t offset, size_t size, const uint8_t *buffer) {
    DevilFSInfo *fs = (DevilFSInfo*)node->fs_private;
    DevilFSInode inode;
    devilfs_read_inode(fs, node->inode, &inode);

    // Journal: begin transaction
    // (simplified - full implementation would write to journal first)

    size_t bytes_written = 0;
    uint8_t block_buf[DEVILFS_BLOCK_SIZE];

    while (bytes_written < size) {
        uint64_t file_block = (offset + bytes_written) / DEVILFS_BLOCK_SIZE;
        uint64_t block_offset = (offset + bytes_written) % DEVILFS_BLOCK_SIZE;

        // Find or allocate block
        uint64_t disk_block = 0;
        uint64_t extent_offset = 0;
        int extent_idx = -1;

        for (int e = 0; e < 12; e++) {
            if (inode.extents[e].length == 0) {
                extent_idx = e;
                break;
            }
            if (file_block < extent_offset + inode.extents[e].length) {
                disk_block = inode.extents[e].block + (file_block - extent_offset);
                break;
            }
            extent_offset += inode.extents[e].length;
        }

        if (disk_block == 0) {
            // Need to allocate new block
            disk_block = devilfs_alloc_blocks(fs, 1);
            if (disk_block == 0) break; // Disk full

            if (extent_idx >= 0) {
                inode.extents[extent_idx].block = disk_block;
                inode.extents[extent_idx].length = 1;
            }
        }

        // Read existing block if partial write
        if (block_offset > 0 || size - bytes_written < DEVILFS_BLOCK_SIZE) {
            devilfs_read_block(fs, disk_block, block_buf);
        }

        // Copy data
        size_t copy_size = DEVILFS_BLOCK_SIZE - block_offset;
        if (copy_size > size - bytes_written) copy_size = size - bytes_written;

        for (size_t i = 0; i < copy_size; i++)
            block_buf[block_offset + i] = buffer[bytes_written + i];

        devilfs_write_block(fs, disk_block, block_buf);
        bytes_written += copy_size;
    }

    // Update inode size
    if (offset + bytes_written > inode.size)
        inode.size = offset + bytes_written;

    // Write inode back
    // (devilfs_write_inode function)

    return bytes_written;
}

// Mount function
static VFSNode* devilfs_mount(const char *device, const char *options) {
    extern void kprintf(const char *fmt, ...);

    // Open block device
    VFSNode *dev = vfs_lookup(device);
    if (!dev) {
        kprintf("[DevilFS] Device not found: %s\n", device);
        return NULL;
    }

    // Read superblock
    DevilFSInfo *fs = (DevilFSInfo*)kzalloc(sizeof(DevilFSInfo));
    uint8_t sb_block[DEVILFS_BLOCK_SIZE];
    vfs_read(dev, 0, DEVILFS_BLOCK_SIZE, sb_block);

    // Copy superblock
    uint8_t *src = sb_block;
    uint8_t *dst = (uint8_t*)&fs->sb;
    for (size_t i = 0; i < sizeof(DevilFSSuperblock); i++)
        dst[i] = src[i];

    // Verify magic
    if (fs->sb.magic != DEVILFS_MAGIC) {
        kprintf("[DevilFS] Invalid magic: 0x%x\n", fs->sb.magic);
        kfree(fs);
        return NULL;
    }

    fs->device = dev;
    fs->encryption_enabled = fs->sb.encryption_enabled;

    // Load bitmaps
    size_t inode_bitmap_size = (fs->sb.total_inodes + 7) / 8;
    size_t block_bitmap_size = (fs->sb.total_blocks + 7) / 8;

    fs->inode_bitmap = (uint8_t*)kmalloc(inode_bitmap_size);
    fs->block_bitmap = (uint8_t*)kmalloc(block_bitmap_size);

    // Read bitmaps from disk
    uint64_t ib_offset = fs->sb.inode_bitmap_block * DEVILFS_BLOCK_SIZE;
    vfs_read(dev, ib_offset, inode_bitmap_size, fs->inode_bitmap);

    uint64_t bb_offset = fs->sb.block_bitmap_block * DEVILFS_BLOCK_SIZE;
    vfs_read(dev, bb_offset, block_bitmap_size, fs->block_bitmap);

    // Create root VFS node
    VFSNode *root = (VFSNode*)kzalloc(sizeof(VFSNode));
    root->name[0] = '/';
    root->type = VFS_DIRECTORY;
    root->inode = fs->sb.root_inode;
    root->permissions = 0755;
    root->ops = &devilfs_file_ops;
    root->fs_private = fs;
    root->ref_count = 1;

    kprintf("[DevilFS] Mounted: %lu blocks, %lu inodes, %lu free blocks\n",
            fs->sb.total_blocks, fs->sb.total_inodes, fs->sb.free_blocks);

    return root;
}

static FilesystemType devilfs_type = {
    .name = "devilfs",
    .mount = devilfs_mount,
};

void devilfs_init(void) {
    vfs_register_filesystem(&devilfs_type);
}

// kernel/drivers/bus/pci.c

#include <devil/types.h>

#define PCI_CONFIG_ADDR     0xCF8
#define PCI_CONFIG_DATA     0xCFC

typedef struct {
    uint16_t vendor_id;
    uint16_t device_id;
    uint8_t  class_code;
    uint8_t  subclass;
    uint8_t  prog_if;
    uint8_t  revision;
    uint8_t  header_type;
    uint8_t  bus, device, function;
    uint32_t bar[6];
    uint8_t  interrupt_line;
    uint8_t  interrupt_pin;
} PCIDevice;

#define MAX_PCI_DEVICES 256
static PCIDevice pci_devices[MAX_PCI_DEVICES];
static int pci_device_count = 0;

static inline void outl(uint16_t port, uint32_t val) {
    __asm__ volatile("outl %0, %1" : : "a"(val), "Nd"(port));
}

static inline uint32_t inl(uint16_t port) {
    uint32_t ret;
    __asm__ volatile("inl %1, %0" : "=a"(ret) : "Nd"(port));
    return ret;
}

uint32_t pci_config_read(uint8_t bus, uint8_t device, uint8_t func, uint8_t offset) {
    uint32_t address = (1 << 31) |                  // Enable
                       ((uint32_t)bus << 16) |
                       ((uint32_t)(device & 0x1F) << 11) |
                       ((uint32_t)(func & 0x07) << 8) |
                       (offset & 0xFC);

    outl(PCI_CONFIG_ADDR, address);
    return inl(PCI_CONFIG_DATA);
}

void pci_config_write(uint8_t bus, uint8_t device, uint8_t func,
                      uint8_t offset, uint32_t value) {
    uint32_t address = (1 << 31) |
                       ((uint32_t)bus << 16) |
                       ((uint32_t)(device & 0x1F) << 11) |
                       ((uint32_t)(func & 0x07) << 8) |
                       (offset & 0xFC);

    outl(PCI_CONFIG_ADDR, address);
    outl(PCI_CONFIG_DATA, value);
}

static void pci_scan_device(uint8_t bus, uint8_t device, uint8_t func) {
    uint32_t reg0 = pci_config_read(bus, device, func, 0x00);
    uint16_t vendor = reg0 & 0xFFFF;
    uint16_t dev_id = (reg0 >> 16) & 0xFFFF;

    if (vendor == 0xFFFF) return; // No device

    PCIDevice *dev = &pci_devices[pci_device_count];
    dev->vendor_id = vendor;
    dev->device_id = dev_id;
    dev->bus = bus;
    dev->device = device;
    dev->function = func;

    uint32_t reg2 = pci_config_read(bus, device, func, 0x08);
    dev->class_code = (reg2 >> 24) & 0xFF;
    dev->subclass = (reg2 >> 16) & 0xFF;
    dev->prog_if = (reg2 >> 8) & 0xFF;
    dev->revision = reg2 & 0xFF;

    uint32_t reg3 = pci_config_read(bus, device, func, 0x0C);
    dev->header_type = (reg3 >> 16) & 0xFF;

    // Read BARs
    for (int i = 0; i < 6; i++) {
        dev->bar[i] = pci_config_read(bus, device, func, 0x10 + i * 4);
    }

    // Read interrupt info
    uint32_t reg15 = pci_config_read(bus, device, func, 0x3C);
    dev->interrupt_line = reg15 & 0xFF;
    dev->interrupt_pin = (reg15 >> 8) & 0xFF;

    extern void kprintf(const char *fmt, ...);
    kprintf("[PCI] %02x:%02x.%x - Vendor: 0x%04x Device: 0x%04x "
            "Class: %02x:%02x\n",
            bus, device, func, vendor, dev_id,
            dev->class_code, dev->subclass);

    pci_device_count++;
}

void pci_init(void) {
    extern void kprintf(const char *fmt, ...);
    kprintf("[PCI] Scanning PCI bus...\n");

    for (int bus = 0; bus < 256; bus++) {
        for (int dev = 0; dev < 32; dev++) {
            uint32_t reg0 = pci_config_read(bus, dev, 0, 0x00);
            if ((reg0 & 0xFFFF) == 0xFFFF) continue;

            pci_scan_device(bus, dev, 0);

            // Check for multi-function device
            uint32_t reg3 = pci_config_read(bus, dev, 0, 0x0C);
            if ((reg3 >> 16) & 0x80) {
                for (int func = 1; func < 8; func++) {
                    reg0 = pci_config_read(bus, dev, func, 0x00);
                    if ((reg0 & 0xFFFF) != 0xFFFF) {
                        pci_scan_device(bus, dev, func);
                    }
                }
            }
        }
    }

    kprintf("[PCI] Found %d devices.\n", pci_device_count);

    // Initialize known drivers based on class codes
    for (int i = 0; i < pci_device_count; i++) {
        PCIDevice *dev = &pci_devices[i];

        switch (dev->class_code) {
            case 0x01: // Mass storage
                if (dev->subclass == 0x06) { // AHCI
                    extern void ahci_init(PCIDevice *dev);
                    ahci_init(dev);
                } else if (dev->subclass == 0x01) { // IDE
                    extern void ata_init(PCIDevice *dev);
                    ata_init(dev);
                } else if (dev->subclass == 0x08) { // NVMe
                    extern void nvme_init(PCIDevice *dev);
                    nvme_init(dev);
                }
                break;

            case 0x02: // Network
                if (dev->vendor_id == 0x8086 && dev->device_id == 0x100E) {
                    extern void e1000_init(PCIDevice *dev);
                    e1000_init(dev);
                } else if (dev->vendor_id == 0x10EC && dev->device_id == 0x8139) {
                    extern void rtl8139_init(PCIDevice *dev);
                    rtl8139_init(dev);
                } else if (dev->vendor_id == 0x1AF4) { // VirtIO
                    extern void virtio_net_init(PCIDevice *dev);
                    virtio_net_init(dev);
                }
                break;

            case 0x03: // Display
                break;

            case 0x04: // Multimedia
                if (dev->subclass == 0x03) { // HD Audio
                    extern void hda_init(PCIDevice *dev);
                    hda_init(dev);
                }
                break;
        }
    }
}

PCIDevice* pci_find_device(uint16_t vendor, uint16_t device) {
    for (int i = 0; i < pci_device_count; i++) {
        if (pci_devices[i].vendor_id == vendor &&
            pci_devices[i].device_id == device) {
            return &pci_devices[i];
        }
    }
    return NULL;
}

PCIDevice* pci_find_class(uint8_t class_code, uint8_t subclass) {
    for (int i = 0; i < pci_device_count; i++) {
        if (pci_devices[i].class_code == class_code &&
            pci_devices[i].subclass == subclass) {
            return &pci_devices[i];
        }
    }
    return NULL;
}
